1 data Arbol a = Vacio | Nodo (Arbol a) a (Arbol a)
    2 
    3 genera = fst $ generaAux 0
    4     where generaAux n = let (l , n1 ) = generaAux (n + 1)
    5                             (r , n2) = generaAux n1
    6                         in (Nodo l n r , n2)
    7 
    8 recorreL (Nodo l x _ ) = x : recorreL l
    9 recorreL Vacio = []
   10 
   11 recorreR (Nodo _ x r ) = x : recorreR r
   12 recorreR Vacio = []
   13 
   14 recorre (Nodo l x r ) = recorre l ++ [x] ++ recorre r
   15 recorre Vacio = []
   16 
   17 take1 _ [] = []
   18 take1 n (x : xs) | n > 0 = x : take1 (n - 1) xs
   19                  | otherwise = []
   20 main = do print (take1 5 . tail . recorreL $ genera)
   21           print (head $ zip (recorreL genera) (recorreR genera))
